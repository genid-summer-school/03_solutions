Background

A unknown disease has been spreading on board of a cruise ship with 2,500 passengers for 21 days. Epidemiologists carry out an investigation of the outbreak and are able to determine some of the characteristics of the new pathogen. They find out that initially, 10 infected persons had boarded the ship and all other passengers never had contact to the pathogen before. Also, once a person gets infected, he or she can immediately transmit the disease to other persons and will stay infectious for 3 days. After that period a person is suspected to have natural immunity lasting around 180 days.

On average, each person had around 5 contacts to other passengers and from the initial analysis, the index case infected on average 40 percent of their contacts (secondary attack rate).

You are asked to do some modeling exercises to fill knowledge gaps and help decision makers to find appropriate measures to contain this outbreak or following outbreaks.

You are given some basic computer code, describing a simple Susceptible-Infected-Recovered model.

library(tidyverse)
library(deSolve)

sir <- function(time, state, parms) {
  # calculate number of populaton size from compartments
  N <- state[1] + state[2] + state[3]
  # calculate the number of new infections
  flow.I <- (state[1] * parms[["beta"]]) * (parms[["contacts"]] %*% (state[2]/N))
  # Increment the number of persons in each compartment
  dS <- -flow.I
  dI <-  flow.I - parms[["gamma"]] * state[2]
  dR <-         + parms[["gamma"]] * state[2] 
  # return result
  return(list(c(dS, dI, dR, flow.I)))
}

##### Transform lsoda-output to data-frame in long-format
run.ode <- function(times, inits, parms, ode.func) { 
  out.ode <- lsoda(y = inits, times = times, func = ode.func, parms = parms) %>%
    as.data.frame() %>% pivot_longer(names_to = "STATE", values_to = "VALUE", !1)
  return(out.ode)
}

plot.ode <- function(df) {
  df <- df %>% filter(STATE != "flow.I")
  p <- ggplot() 
  p <- p + geom_line(data = df, aes(x=time, y = VALUE, color=STATE)) 
  p <- p + theme_bw() + theme(legend.position = "bottom")
  p
}

sirs <- function(time, state, parms) {
  # calculate number of populaton size from compartments
  N <- state[1] + state[2] + state[3]

sirs <- function(time, state, parms) {
  # calculate number of populaton size from compartments
  N <- state[1] + state[2] + state[3]
  # calculate the number of new infections
  flow.I <- (state[1] * parms[["beta"]] *(1 + parms[["tau"]] * cos(2*pi*(time-7)/14))) * (parms[["contacts"]] %*% (state[2]/N))
  # Increment the number of persons in each compartment

}

# solution 1 -- death at infection
sirs <- function(time, state, parms) {
  # calculate number of populaton size from compartments
  N <- state[1] + state[2] + state[3]
  # calculate the number of new infections
  flow.I <- (state[1] * parms[["beta"]]) * (parms[["contacts"]] %*% (state[2]/N))
  # Increment the number of persons in each compartment
  dS <- -flow.I                                                     + parms[["sigma"]] * state[3]
  dI <-  flow.I * (1 - parms[["xi"]]) - parms[["gamma"]] * state[2]
  dR <-                               + parms[["gamma"]] * state[2] - parms[["sigma"]] * state[3]
  dD <- + flow.I * parms[["xi"]]
  # return result
  return(list(c(dS, dI, dR, dD, flow.I)))
}
